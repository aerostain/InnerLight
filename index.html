ds <- diamonds
idx <- sample(seq(ds %>% dim() %>% .[1]), 5000,replace = TRUE)
ds %<>% .[idx, ]

gr1 <-
  ds %>%
  ggplot(aes(cut, price,group = 1)) +
  stat_summary(
    geom = "errorbar",
    fun.data = mean_cl_normal,
    width = .2, col = "blue"
  ) +
  stat_summary(
    geom = "line",
    fun.data = mean_cl_normal,
    col = "red", linetype = 2
  ) +
  dark_mode()

ggplotly(gr1)
